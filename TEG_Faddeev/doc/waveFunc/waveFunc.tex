\title{Longitudinal response function of Deuteron}

\begin{document}
\begin{equation}
  \begin{aligned}
  \braket{\psi|\psi} 
  = \sum_{\alpha}\int{p^2dp}\int{q^2dq}\braket{\psi|\alpha, p, q} \braket{\alpha, p, q| \psi}
  = 1
  \end{aligned}
\end{equation}
